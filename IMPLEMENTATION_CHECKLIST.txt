# ==================== 1. GET ADMIN METRICS ====================

@router.get("/metrics")
async def get_admin_metrics(
    db_session: SessionDep,
    current_admin: CurrentAdminUserDep
):
    """
    Returns aggregate statistics for stat cards.
    
    Response (200 OK):
    {
        "total_users": 1542,
        "active_users": 1203,
        "kyc_verified": 1089,
        "account_holds": 12
    }
    """
    try:
        # Total users
        total_users_result = await db_session.execute(
            select(func.count(DBUser.id))
        )
        total_users = total_users_result.scalar() or 0

        # Active users (logged in within 30 days)
        thirty_days_ago = datetime.utcnow() - timedelta(days=30)
        active_users_result = await db_session.execute(
            select(func.count(DBUser.id)).where(
                DBUser.updated_at >= thirty_days_ago
            )
        )
        active_users = active_users_result.scalar() or 0

        # KYC verified
        kyc_verified_result = await db_session.execute(
            select(func.count(DBUser.id)).where(
                DBUser.kyc_status == "approved"
            )
        )
        kyc_verified = kyc_verified_result.scalar() or 0

        # Account holds
        holds_result = await db_session.execute(
            select(func.count(AccountHold.id)).where(
                AccountHold.released_at.is_(None)
            )
        )
        account_holds = holds_result.scalar() or 0

        return {
            "total_users": total_users,
            "active_users": active_users,
            "kyc_verified": kyc_verified,
            "account_holds": account_holds
        }

    except Exception as e:
        logger.error(f"metrics error: {e}")
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=str(e)
        )


# ==================== 2. LIST USERS WITH PAGINATION ====================

@router.get("/users")
async def list_users(
    db_session: SessionDep,
    current_admin: CurrentAdminUserDep,
    skip: int = Query(0, ge=0),
    limit: int = Query(20, ge=1, le=100)
):
    """
    Get paginated user list with all required fields.
    
    Response (200 OK):
    {
        "data": [...],
        "total_users": 1542,
        "total_pages": 78,
        "current_page": 1
    }
    """
    try:
        # Get total count
        total_result = await db_session.execute(
            select(func.count(DBUser.id))
        )
        total_users = total_result.scalar() or 0

        # Calculate pagination
        total_pages = math.ceil(total_users / limit) if total_users > 0 else 1
        current_page = (skip // limit) + 1

        # Get users
        users_result = await db_session.execute(
            select(DBUser)
            .order_by(desc(DBUser.created_at))
            .offset(skip)
            .limit(limit)
        )
        users = users_result.scalars().all()

        # Format response
        user_data = []
        for user in users:
            # Count holds for this user
            holds_result = await db_session.execute(
                select(func.count(AccountHold.id)).where(
                    AccountHold.user_id == user.id,
                    AccountHold.released_at.is_(None)
                )
            )
            account_holds_count = holds_result.scalar() or 0

            user_data.append({
                "id": user.id,
                "email": user.email,
                "full_name": user.full_name,
                "is_active": user.is_active,
                "kyc_status": user.kyc_status,
                "credit_score": getattr(user, 'credit_score', 0),
                "account_holds_count": account_holds_count,
                "phone_number": getattr(user, 'phone_number', ''),
                "last_login": user.updated_at.isoformat() if user.updated_at else None
            })

        return {
            "data": user_data,
            "total_users": total_users,
            "total_pages": total_pages,
            "current_page": current_page
        }

    except Exception as e:
        logger.error(f"list_users error: {e}")
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=str(e)
        )

# ==================== 7. CREATE HOLD ====================

@router.post("/intervention/holds/create")
async def create_hold(
    request: CreateHoldRequest,
    db_session: SessionDep,
    current_admin: CurrentAdminUserDep
):
    """
    Create a new account hold on a user.
    
    Response (201 Created):
    {
        "id": 102,
        "user_id": 42,
        "hold_type": "COMPLIANCE",
        "reason": "Detailed reason for hold",
        "created_at": "2025-04-08T14:32:15Z",
        "release_date": "2025-05-08T14:32:15Z"
    }
    """
    try:
        # Check user exists
        user_result = await db_session.execute(
            select(DBUser).where(DBUser.id == request.user_id)
        )
        user = user_result.scalar_one_or_none()
        if not user:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail="User not found"
            )

        # Check for existing active hold
        existing_hold_result = await db_session.execute(
            select(AccountHold).where(
                AccountHold.user_id == request.user_id,
                AccountHold.released_at.is_(None)
            )
        )
        existing_hold = existing_hold_result.scalar_one_or_none()
        if existing_hold:
            raise HTTPException(
                status_code=status.HTTP_422_UNPROCESSABLE_ENTITY,
                detail="User already has an active hold"
            )

        # Create hold
        release_date = datetime.utcnow() + timedelta(days=request.duration_days)
        
        new_hold = AccountHold(
            user_id=request.user_id,
            hold_type=request.hold_type,
            reason=request.reason,
            current_reason=request.reason,
            release_date=release_date,
            created_by=request.created_by
        )

        db_session.add(new_hold)
        await db_session.commit()
        await db_session.refresh(new_hold)

        # Log action
        audit_log = AdminAuditLog(
            admin_id=current_admin.id,
            admin_email=current_admin.email,
            action="HOLD_CREATE",
            target_user_id=request.user_id,
            details=f"Hold Type: {request.hold_type}, Duration: {request.duration_days} days",
            created_at=datetime.utcnow()
        )
        db_session.add(audit_log)
        await db_session.commit()

        return {
            "id": new_hold.id,
            "user_id": new_hold.user_id,
            "hold_type": new_hold.hold_type,
            "reason": new_hold.reason,
            "created_at": new_hold.created_at.isoformat() if new_hold.created_at else None,
            "release_date": release_date.isoformat()
        }

    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"create_hold error: {e}")
        await db_session.rollback()
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=str(e)
        )

@router.post("/kyc/{user_id}/approve")
async def approve_kyc(
    user_id: int,
    request: ApproveKYCRequest,
    db_session: SessionDep,
    current_admin: CurrentAdminUserDep
):
    """
    Approve KYC for a user.
    
    Response (200 OK):
    {
        "user_id": 42,
        "kyc_status": "VERIFIED",
        "approved_at": "2025-04-08T14:32:15Z"
    }
    """
    try:
        # Get user
        user_result = await db_session.execute(
            select(DBUser).where(DBUser.id == user_id)
        )
        user = user_result.scalar_one_or_none()
        if not user:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail="User not found"
            )

        # Update user KYC status
        user.kyc_status = "approved"
        user.updated_at = datetime.utcnow()

        # Update KYC submission if exists
        kyc_result = await db_session.execute(
            select(KYCSubmission)
            .where(KYCSubmission.user_id == user_id)
            .order_by(desc(KYCSubmission.submitted_at))
        )
        kyc = kyc_result.scalars().first()
        if kyc:
            kyc.status = "approved"
            kyc.approved_at = datetime.utcnow()
            db_session.add(kyc)

        db_session.add(user)
        await db_session.commit()

        # Log action
        audit_log = AdminAuditLog(
            admin_id=current_admin.id,
            admin_email=current_admin.email,
            action="KYC_APPROVAL",
            target_user_id=user_id,
            details="KYC status changed to APPROVED",
            created_at=datetime.utcnow()
        )
        db_session.add(audit_log)
        await db_session.commit()

        return {
            "user_id": user_id,
            "kyc_status": "VERIFIED",
            "approved_at": datetime.utcnow().isoformat()

):

    Reject KYC for a user and send notification email.

# ==================== 14. GLOBAL SEARCH ====================

@router.get("/intervention/search")
async def global_search(
    db_session: SessionDep,
    current_admin: CurrentAdminUserDep,
    query: str = Query(..., min_length=1),
    limit: int = Query(20, ge=1, le=100)
):
    """
    Global search across users and accounts.
    
    Response (200 OK):
    {
        "results": {
            "users": [...],
            "accounts": [...]
        }
    }
    """
    try:
        results = {"users": [], "accounts": []}

        # Search users by email or name
        users_result = await db_session.execute(
            select(DBUser)
            .where(
                or_(
                    DBUser.email.ilike(f"%{query}%"),
                    DBUser.full_name.ilike(f"%{query}%")
                )
            )
            .limit(limit)
        )
        users = users_result.scalars().all()
        results["users"] = [
            {
                "id": u.id,
                "email": u.email,
                "full_name": u.full_name
            }
            for u in users
        ]

        # Search accounts by account number
        accounts_result = await db_session.execute(
            select(DBAccount)
            .where(DBAccount.account_number.ilike(f"%{query}%"))
            .limit(limit)
        )
        accounts = accounts_result.scalars().all()
        results["accounts"] = [
            {
                "id": a.id,
                "account_number": a.account_number
            }
            for a in accounts
        ]

        return {"results": results}

    except Exception as e:
        logger.error(f"global_search error: {e}")
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=str(e)
        )

@router.get("/intervention/activity-feed")
async def get_activity_feed(
    db_session: SessionDep,
    current_admin: CurrentAdminUserDep,
    limit: int = Query(20, ge=1, le=100),
    hours_back: int = Query(24, ge=1, le=720)
):
    """
    Get recent activity for real-time feed.
    
    Response (200 OK):
    {
        "feed_items": [
            {
                "type": "transaction",
                "description": "User John Doe deposited $500",
                "timestamp": "2025-04-08T14:32:15Z"
            }
        ]
    }
    """
    try:
        time_filter = datetime.utcnow() - timedelta(hours=hours_back)
        feed_items = []

        # Recent transactions
        txn_result = await db_session.execute(
            select(Transaction)
            .where(Transaction.created_at >= time_filter)
            .order_by(desc(Transaction.created_at))
            .limit(limit)
        )
        transactions = txn_result.scalars().all()

        for txn in transactions:
            feed_items.append({
                "type": "transaction",
                "description": f"Transaction: {txn.transaction_type} - ${txn.amount}",
                "timestamp": txn.created_at.isoformat() if txn.created_at else None
            })

        return {"feed_items": feed_items[:limit]}

    except Exception as e:
        logger.error(f"get_activity_feed error: {e}")
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=str(e)
        )

    """
    Send bulk email to users based on KYC status.
    
    Response (200 OK):
    {
        "campaign_id": "camp_abc123",

            status_filter = "rejected"

            status_filter = None

        if status_filter:
            users_result = await db_session.execute(
                select(DBUser).where(DBUser.kyc_status == status_filter)
            )
        else:
            users_result = await db_session.execute(select(DBUser))

        users = users_result.scalars().all()
        recipient_count = len(users)

        # Generate campaign ID
        campaign_id = f"camp_{datetime.utcnow().timestamp()}"

        # Log action
        audit_log = AdminAuditLog(
            admin_id=current_admin.id,
            admin_email=current_admin.email,
            action="BULK_EMAIL_SENT",
            details=f"Campaign {campaign_id}: {recipient_count} recipients, Filter: {request.kyc_status_filter}",
            created_at=datetime.utcnow()
        )
        db_session.add(audit_log)
        await db_session.commit()

        return {
            "campaign_id": campaign_id,
            "recipient_count": recipient_count,
            "status": "QUEUED",
            "sent_at": datetime.utcnow().isoformat()
        }

    except Exception as e:
        logger.error(f"bulk_email error: {e}")
        await db_session.rollback()
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,

# ==================== 18. DELETE USER (SUPERADMIN ONLY) ====================

@router.delete("/users/{user_id}")
async def delete_user(
    user_id: int,
    db_session: SessionDep,
    current_admin: CurrentAdminUserDep
):
    """
    Delete a user (SUPERADMIN only).
    
    Response (200 OK):
    {
        "id": 42,
        "status": "DELETED",
        "deleted_at": "2025-04-08T14:32:15Z"
    }
    """
    try:
        # Check permissions
        if not getattr(current_admin, 'is_superadmin', False):
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail="Requires SUPER_ADMIN role"
            )

        # Get user
        user_result = await db_session.execute(
            select(DBUser).where(DBUser.id == user_id)
        )
        user = user_result.scalar_one_or_none()
        if not user:
            raise HTTPException(
                status_code=status.HTTP_404_NOT_FOUND,
                detail="User not found"
            )

        # Check for active holds
        holds_result = await db_session.execute(
            select(func.count(AccountHold.id)).where(
                AccountHold.user_id == user_id,
                AccountHold.released_at.is_(None)
            )
        )
        active_holds_count = holds_result.scalar() or 0
        if active_holds_count > 0:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="Cannot delete user with active holds"
            )

        # Mark user as deleted
        user.is_active = False
        user.updated_at = datetime.utcnow()
        db_session.add(user)

        # Log action
        audit_log = AdminAuditLog(
            admin_id=current_admin.id,
            admin_email=current_admin.email,
            action="USER_DELETED",
            target_user_id=user_id,
            details=f"User {user.email} deleted by superadmin",
            created_at=datetime.utcnow()
        )
        db_session.add(audit_log)
        await db_session.commit()

        return {
            "id": user_id,
            "status": "DELETED",
            "deleted_at": datetime.utcnow().isoformat()
        }

    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"delete_user error: {e}")
        await db_session.rollback()
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=str(e)
        )
